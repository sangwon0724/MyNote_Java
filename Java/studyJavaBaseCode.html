기초 (출력 및 입력)
  • 출력 
                    public class Main {
                    public static void main(String[] args)
                    {
                        System.out.println("출력");
                    }
                }
  • 입력 
                    import java.util.*;
                public class Main{
                    public static void main(String args[]){
                        //주의사항
                        //1. int는 여러줄로 받아도 된다.
                        //2. 문자를 받을 경우 엔터를 누르면 \n이 들어가게 되니 조심해야 한다.
                        //3. char형은 next()나 nextLine()으로 받을수 없다.
                        Scanner sc = new Scanner(System.in);

                        //int를 입력받는 경우 - Scanner 사용
                        int num;
                        num = sc.nextInt();
                        System.out.println("int형 변수 num : " + num);

                        
                        //String을 입력받는 경우 - Scanner 사용
                        String str;
                        str = sc.nextLine();
                        System.out.println("String형 변수 str : " + str);

                        sc.close(); //스캐너 종료
                    }
                }
