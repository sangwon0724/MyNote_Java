기초 (출력 및 입력)
  • 출력 
                    public class Main {
                    public static void main(String[] args)
                    {
                        System.out.println("출력");
                    }
                }
  • 입력 
                    import java.util.*;
                public class Main{
                    public static void main(String args[]){
                        Scanner sc = new Scanner(System.in);
                        int a, b;
                        a = sc.nextInt();
                        b = sc.nextInt();
                        System.out.println(a + b);
                    }
                }
